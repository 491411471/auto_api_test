SQL: WITH order_counts AS (
SELECT 
order_id,
COUNT(*) OVER (PARTITION BY order_id) as stage_count,
ROW_NUMBER() OVER (PARTITION BY order_id ORDER BY create_time DESC) as rn
FROM llxz_order.ct_order_by_stages
WHERE shop_id = '71008738021cd3393bacbac182bd6a86af0b5c87')
SELECT order_id
FROM order_counts
WHERE stage_count > 1 AND rn = 1
LIMIT 1;
